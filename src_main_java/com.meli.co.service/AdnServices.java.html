package com.meli.co.service;

import java.util.HashMap;
import java.util.Map;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.meli.co.dtos.AdnDto;
import com.meli.co.model.Adn;
import com.meli.co.repository.AdnRepository;


@Service
public class AdnServices {
	@Autowired
	private AdnRepository adnRepository;

	public Map<String,Double> findAll(){
		Map<String, Double> stadists = new HashMap<>();
		double humanos  =adnRepository.findAll().size();
		double mutante = adnRepository.getIsMutante(true);
		double radioModulos =  mutante/humanos;
		stadists.put("count_mutant_dna",mutante );
		stadists.put("count_humanos_dna",  humanos);
		stadists.put("ratio", radioModulos);
		return stadists;
	}

	public boolean transaccion(AdnDto adnDto){
		String sangre = converterString( adnDto.getDna());
		boolean isMutante =false;
		if (sangre.matches("[ATCG]+")) {
			boolean exists = existsRegister(sangre);
			if (!exists) {
				isMutante = isMutante(adnDto.getDna());
				save(sangre,isMutante);
			}
		}
		return isMutante;
	}


	public String converterString(String [] sangre) {
		StringBuilder cadena = new StringBuilder();
		for (int x=0;x<sangre.length;x++){
			cadena =cadena.append(sangre[x]);
		}
		return cadena.toString();
	}

	public Adn save(String  sangre ,boolean isMutante) {
		Adn adn = new Adn();
		adn.setName(sangre);
		adn.setMutante(isMutante);

		return adnRepository.saveAndFlush(adn);
	}
	public boolean existsRegister(String vectorAdn) {
		Optional<Adn> optionalHumano = adnRepository.findByName(vectorAdn);
		return optionalHumano.isPresent();

	}

	public boolean isMutante(String[] vectorAdn) {

		byte secuenciasEncuentradas = 0;
		char []  bldmatrizDiagonal = new char[vectorAdn.length];
		for (int i=0; i < vectorAdn.length; i++) {
			char [] bldmatrizIzquierdaDerecha = new char[vectorAdn.length];
			char [] bldmatrizArribaAbajo = new char[vectorAdn.length];
			if(vectorAdn[i].length()==vectorAdn.length) {
				for (int j = 0; j <vectorAdn.length; j ++) {

					bldmatrizArribaAbajo[j]=(vectorAdn[i].charAt (j));	
					bldmatrizIzquierdaDerecha[j]=(vectorAdn[j].charAt (i));	
					if (i==j) {
						bldmatrizDiagonal[i]=(vectorAdn[i].charAt (j));
					}
					
				}
				secuenciasEncuentradas+=possibleMutante(bldmatrizArribaAbajo);
				secuenciasEncuentradas+=possibleMutante(bldmatrizIzquierdaDerecha);
				if (secuenciasEncuentradas>=2) {
					break;
				}	
			}
			else {
				return false;
			}

		}

		secuenciasEncuentradas+=possibleMutante(bldmatrizDiagonal);

		return secuenciasEncuentradas>=2;
	}

	/**
	 * 
	 * @param aux
	 * @return
	 */
	private byte possibleMutante(char[] aux ) {
		char inicial = aux[0];
		byte contadorNitrogeno = 0;
		byte posibleMutante =0;
		for (int x = 1; x <aux.length; x ++) {
			char secuencia = aux[x];
			if (inicial==secuencia) {
				contadorNitrogeno ++;
				if (contadorNitrogeno==3) {
					posibleMutante ++;
					inicial ='9';
					contadorNitrogeno=1;
				}	
			}else {
				inicial= secuencia;	
				contadorNitrogeno= 0;
			}
		}


		return posibleMutante ;
	}

}
